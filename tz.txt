python# backend/auth.py
from fastapi import Depends, HTTPException
from jose import JWTError, jwt

def get_current_user(token: str = Depends(oauth2_scheme)):
    # Har ikki modul uchun umumiy auth
    credentials_exception = HTTPException(
        status_code=401,
        detail="Could not validate credentials"
    )
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        user_id: str = payload.get("sub")
        # User qaysi modulga kirishi mumkinligini tekshir
        allowed_modules = payload.get("modules", [])
        return {"user_id": user_id, "modules": allowed_modules}
    except JWTError:
        raise credentials_exception

python# backend/billing.py
class BillingService:
    def calculate_monthly_cost(self, user_id: str):
        """Calculate cost based on usage across all modules"""
        bankcall_calls = get_bankcall_usage(user_id)
        antifraud_transactions = get_antifraud_usage(user_id)
        
        cost = {
            "bankcall": bankcall_calls * 0.05,  # $0.05 per call
            "antifraud": antifraud_transactions * 0.02,  # $0.02 per transaction
            "total": 0
        }
        cost["total"] = sum(cost.values())
        return cost

5.1. Users & Organizations
sql-- Users table
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(255) NOT NULL,
    phone VARCHAR(20),
    role VARCHAR(50) NOT NULL, -- super_admin, org_admin, manager, viewer
    organization_id UUID REFERENCES organizations(id),
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Organizations table
CREATE TABLE organizations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    type VARCHAR(50), -- bank, mfi, fintech
    modules JSONB NOT NULL, -- ["bankcall", "antifraud"]
    subscription_tier VARCHAR(50), -- starter, business, enterprise
    billing_email VARCHAR(255),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);
5.2. BankCall AI
sql-- Calls table
CREATE TABLE bankcall_calls (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID REFERENCES organizations(id),
    call_id VARCHAR(100) UNIQUE NOT NULL,
    phone_number VARCHAR(20) NOT NULL,
    scenario_id VARCHAR(100),
    status VARCHAR(50), -- initiated, in_progress, completed, failed
    duration_seconds INT,
    transcript TEXT,
    customer_satisfaction DECIMAL(2,1),
    cost DECIMAL(10,2),
    created_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP
);

-- Scenarios table
CREATE TABLE bankcall_scenarios (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID REFERENCES organizations(id),
    scenario_id VARCHAR(100) UNIQUE NOT NULL,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    config JSONB NOT NULL, -- Full scenario configuration
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);
5.3. Anti-Fraud AI
sql-- Transactions table
CREATE TABLE antifraud_transactions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID REFERENCES organizations(id),
    transaction_id VARCHAR(100) UNIQUE NOT NULL,
    user_id VARCHAR(100) NOT NULL,
    amount DECIMAL(15,2) NOT NULL,
    currency VARCHAR(10) NOT NULL,
    merchant VARCHAR(255),
    merchant_category VARCHAR(100),
    ip_address VARCHAR(45),
    location JSONB, -- {country, city, lat, lon}
    device_id VARCHAR(255),
    decision VARCHAR(50), -- APPROVED, UNDER_REVIEW, BLOCKED
    risk_score INT,
    confidence DECIMAL(3,2),
    reasons JSONB,
    ml_model_version VARCHAR(50),
    processing_time_ms INT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Fraud Rules table
CREATE TABLE antifraud_rules (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id UUID REFERENCES organizations(id),
    rule_id VARCHAR(100) UNIQUE NOT NULL,
    name VARCHAR(255) NOT NULL,

name: AI Foundry CI/CD

on:
  push:

      - uses: actions/checkout@v3
      
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'
      
      - name: Install dependencies
        run: |
          pip install -r backend/requirements.txt
          pip install pytest pytest-cov
      
      - name: Run tests
        run: |
          cd backend
          pytest --cov=. --cov-report=xml
      
      - name: Upload coverage
        uses: codecov/codecov-action@v3

  build:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Build Docker images
        run: |
          docker build -t aifoundry/bankcall:${{ github.sha }} ./backend/bankcall
          docker build -t aifoundry/antifraud:${{ github.sha }} ./backend/antifraud
      
      - name: Push to registry
        run: |
          echo ${{ secrets.DOCKER_PASSWORD }} | docker login -u ${{ secrets.DOCKER_USERNAME }} --password-stdin
          docker push aifoundry/bankcall:${{ github.sha }}
          docker push aifoundry/antifraud:${{ github.sha }}

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - name: Deploy to Kubernetes
        run: |
          kubectl set image deployment/bankcall-service bankcall-api=aifoundry/bankcall:${{ github.sha }} -n ai-foundry-prod
          kubectl set image deployment/antifraud-service antifraud-api=aifoundry/antifraud:${{ github.sha }} -n ai-foundry-prod
          kubectl rollout status deployment/bankcall-service -n ai-foundry-prod
          kubectl rollout status deployment/antifraud-service -n ai-foundry-prod
